rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{userId} {

      match /reminders/{reminderId} {
        allow read, delete: if request.auth != null && request.auth.uid == userId;
        // Client may only create/update the fields it's allowed to set;
        // "sent" is reset to false by the client on create but only the
        // Cloud Function (Admin SDK, bypasses rules) should ever flip it
        // back to true.
        allow create, update: if request.auth != null
          && request.auth.uid == userId
          && request.resource.data.sent == false;
      }

      match /fcmTokens/{tokenId} {
        allow read, write, delete: if request.auth != null && request.auth.uid == userId;
      }
    }

    // Deny everything else by default.
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
